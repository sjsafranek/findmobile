# INSTALL CORDOVA AND PLUGMAN
npm install -g cordova
npm install -g plugman

# CREATE PROJECT
cordova create hello com.example.hello HelloWorld

# CD INTO PROJECT DIRECTORY
cd hello

# ADD PLATFORMS
cordova platform add android

# CHECK PLATFORMS
cordova platforms ls

# SET ENVIRONMENTAL VARIABLES (WINDOWS)
set ANDROID_HOME=C:\\Users\Stefan\Documents\Programs\java\adt-bundle-windows-x86_64-20130514\sdk
set PATH=%PATH%;%ANDROID_HOME%\tools;%ANDROID_HOME%\platform-tools

# BUILD PROJECT
cordova build

cordova run android



# WIFISCANNER
cordova plugin add nl.nielsad.cordova.wifiscanner
